# gitwin - git windows for node.js

### example - pull
```
	var _gitwin = require('gitwin');
	var gitwin = new _gitwin(your_callback);
	gitwin.path = "c:/your_app";
	gitwin.pull();
};
```

### example - status
```
	var _gitwin = require('gitwin');
	var gitwin = new _gitwin(your_callback);
	gitwin.path = "c:/your_app";
	gitwin.status();
};
```
